<Callout type="tip" title="Zero-friction setup">
Use one installer command per runtime. Each installer handles platform detection, binary install, and supported MCP config merge.
</Callout>

## Prereqs

- macOS, Linux, or Windows with WSL
- `curl` available
- Rust toolchain installed (fallback source build path)

<Steps>
  <Step number={1} title="Install AgenticMemory">
    <Command
      title="Memory runtime"
      description="Installs amem + memory MCP server and applies compatible local client config updates."
      code={`curl -fsSL https://agentralabs.tech/install/memory | bash`}
    />
    <CliOutput
      title="Expected output"
      output={`[ok] Platform detected\n[info] Configuring MCP clients...\n[ok] Done! Memory defaults to ~/.brain.amem`}
    />
  </Step>

  <Step number={2} title="Install AgenticVision">
    <Command
      title="Vision runtime"
      description="Installs vision MCP runtime for browserless visual workflows."
      code={`curl -fsSL https://agentralabs.tech/install/vision | bash`}
    />
  </Step>

  <Step number={3} title="Install AgenticCodebase">
    <Command
      title="Codebase runtime"
      description="Installs acb + acb-mcp for semantic code intelligence workflows."
      code={`curl -fsSL https://agentralabs.tech/install/codebase | bash`}
    />
  </Step>

  <Step number={4} title="Verify binaries">
    <Command
      title="Version checks"
      description="Confirm each binary is available in your shell PATH."
      code={`amem --version\nagentic-vision-mcp --version\nacb --version\nacb-mcp --version`}
    />
  </Step>

  <Step number={5} title="Optional MCP runtime checks">
    <Command
      title="Runtime sanity"
      description="Useful when validating desktop client integration and troubleshooting startup issues."
      code={`agentic-memory-mcp --help\nagentic-vision-mcp serve --help\nacb-mcp serve --help`}
    />
  </Step>
</Steps>
