<Callout type="tip" title="Zero-friction setup">
Use one installer command per runtime. Each installer handles platform detection, binary install, and supported MCP config merge.
</Callout>

## Prereqs

- macOS, Linux, or Windows with WSL
- `curl` available
- Rust toolchain installed (fallback source build path)

<Steps>
  <Step number={1} title="Install one runtime (or all three)">
    <CommandTabs
      title="Hosted install channels"
      description="Install independently by project. Each installer applies platform detection and safe defaults."
      items={[
        {
          label: "AgenticMemory",
          code: "curl -fsSL https://agentralabs.tech/install/memory | bash",
          description: "Installs amem + agentic-memory-mcp and applies compatible client merge behavior.",
        },
        {
          label: "AgenticVision",
          code: "curl -fsSL https://agentralabs.tech/install/vision | bash",
          description: "Installs agentic-vision-mcp for browserless perception and visual continuity workflows.",
        },
        {
          label: "AgenticCodebase",
          code: "curl -fsSL https://agentralabs.tech/install/codebase | bash",
          description: "Installs acb + acb-mcp for semantic code intelligence and MCP operations.",
        },
      ]}
    />
  </Step>

  <Step number={2} title="Run binary checks">
    <Command
      title="Version checks"
      description="Confirm installed binaries are available in your current PATH."
      code={`amem --version
agentic-memory-mcp --version
agentic-vision-mcp --version
acb --version
acb-mcp --version`}
    />
  </Step>

  <Step number={3} title="Validate MCP entrypoints">
    <Command
      title="Runtime sanity"
      description="Use this output pattern to verify MCP entrypoints before connecting desktop clients."
      code={`agentic-memory-mcp --help
agentic-vision-mcp serve --help
acb-mcp serve --help`}
    />
    <CliOutput
      title="Expected signal"
      output={`[ok] Platform detected
[ok] Runtime installed
[info] MCP client merge completed (where supported)
[ok] Entry-point command responds without error`}
    />
  </Step>
</Steps>
